```rust
#![cfg_attr(not(debug_assertions), windows_subsystem = "windows")]

use tauri::Manager;

fn main() {
    tauri::Builder::default()
        .setup(|app| {
            let main_window = app.get_window("main").unwrap();
            
            // 窗口设置
            main_window.set_decorations(false)?;      // 无边框
            main_window.set_always_on_top(true)?;     // 始终置顶
            main_window.set_transparent(true)?;       // 透明背景
            
            // 系统托盘
            let tray_id = "pet-tray";
            SystemTray::new()
                .with_id(tray_id)
                .with_menu(
                    SystemTrayMenu::new()
                        .add_item(CustomMenuItem::new("show", "显示宠物"))
                        .add_item(CustomMenuItem::new("hide", "隐藏宠物"))
                        .add_native_item(SystemTrayMenuItem::Separator)
                        .add_item(CustomMenuItem::new("settings", "设置"))
                        .add_native_item(SystemTrayMenuItem::Separator)
                        .add_item(CustomMenuItem::new("quit", "退出"))
                )
                .build(app)?;
            
            Ok(())
        })
        .on_system_tray_event(|app, event| {
            match event {
                SystemTrayEvent::MenuItemClick { id, .. } => {
                    match id.as_str() {
                        "hide" => app.get_window("main").unwrap().hide().unwrap(),
                        "show" => app.get_window("main").unwrap().show().unwrap(),
                        "quit" => app.exit(0),
                        _ => {}
                    }
                }
            }
        })
        .run(tauri::generate_context!())
        .expect("error while running tauri application");
}
```

### 前端拖拽实现

```typescript
// src/composables/useDrag.ts
export function useDrag(element: Ref<HTMLElement | null>) {
  let startX = 0, startY = 0
  
  const onMouseDown = (e: MouseEvent) => {
    startX = e.clientX
    startY = e.clientY
    document.addEventListener('mousemove', onMouseMove)
    document.addEventListener('mouseup', onMouseUp)
  }
  
  const onMouseMove = (e: MouseEvent) => {
    const deltaX = e.clientX - startX
    const deltaY = e.clientY - startY
    // 调用Tauri API移动窗口
    invoke('move_window', { deltaX, deltaY })
  }
  
  return { onMouseDown }
}
```

### 验收标准

- [ ] 启动后宠物显示在桌面右下角，无窗口边框
- [ ] 拖拽宠物可移动到任意位置
- [ ] 鼠标滚轮在宠物上滑动 → 宠物大小不变化（禁用滚轮缩放）
- [ ] 右键点击宠物 → 弹出菜单（隐藏 / 设置 / 退出）

- [ ] 右键菜单点"设置" → 打开独立设置窗口（大小适中）
- [ ] 右键菜单点"退出" → 进程完全终止
- [ ] 设置窗口拖动滑块 → 宠物大小实时变化（10%~100%范围）
- [ ] 重启应用后宠物大小保持上次设置
- [ ] 右键系统托盘图标，点"显示" → 宠物重新出现

### 状态机设计

```typescript
// 基础状态
type BaseState = 'sitting' | 'sleeping'

// 临时动作/过渡状态
type TempAction = 
  | 'talking' | 'happy'              // 基础互动
  | 'tilt-head' | 'wag-tail'        // 点击互动
  | 'lying-up'                       // 从睡眠恢复过渡
  | 'crazy' | 'crazy-plus'           // 兴奋状态
  | 'running' | 'backing'            // 超兴奋与恢复
  | 'bored'                          // 无聊状态

interface StateMachine {
  baseState: BaseState                // 当前基础状态
  currentAction: TempAction | null    // 当前临时动作
  actionQueue: TempAction[]           // 待执行动作队列
  
  // 空闲计时
  idleStartTime: number                // 空闲开始时间戳
  lastActivityTime: number             // 最后活动时间
  
  // 点击计数
  clickCount: number                   // 连续点击计数
  firstClickTime: number               // 第一次点击时间
  lastClickTime: number                // 最后一次点击时间
  
  // 特殊状态计时
  runningStartTime: number             // 开始running的时间
}
```

### 核心代码

```typescript
// src/composables/usePetState.ts
import { ref, computed, onMounted, onUnmounted } from 'vue'
import { invoke } from '@tauri-apps/api/tauri'

// ==================== 类型定义 ====================
type BaseState = 'sitting' | 'sleeping'
type TempAction = 
  | 'talking' | 'happy' 
  | 'tilt-head' | 'wag-tail' 
  | 'lying-up'
  | 'crazy' | 'crazy-plus' 
  | 'running' | 'backing'
  | 'bored'

// ==================== GIF资源映射 ====================
const PETS = {
  dog: {
    // 基础状态
    sitting: new URL('../assets/pets/dog/sitting.gif', import.meta.url).href,
    sleeping: new URL('../assets/pets/dog/sleeping.gif', import.meta.url).href,
    
    // 互动动作
    'tilt-head': new URL('../assets/pets/dog/tilt-head.gif', import.meta.url).href,
    'wag-tail': new URL('../assets/pets/dog/wag-tail.gif', import.meta.url).href,
    happy: new URL('../assets/pets/dog/happy.gif', import.meta.url).href,
    talking: new URL('../assets/pets/dog/talking.gif', import.meta.url).href,
    bored: new URL('../assets/pets/dog/bored.gif', import.meta.url).href,
    
    // 过渡动画
    'lying-up': new URL('../assets/pets/dog/back/sitting.gif', import.meta.url).href,
    backing: new URL('../assets/pets/dog/backing.gif', import.meta.url).href,
    
    // 兴奋状态
    crazy: new URL('../assets/pets/dog/crazy.gif', import.meta.url).href,
    'crazy-plus': new URL('../assets/pets/dog/crazy-plus.gif', import.meta.url).href,
    running: new URL('../assets/pets/dog/running.gif', import.meta.url).href,
  }
}

// ==================== 状态变量 ====================
const currentPet = ref('dog')
const baseState = ref<BaseState>('sitting')
const currentAction = ref<TempAction | null>(null)
const actionQueue = ref<{ name: TempAction; duration: number; priority: number; loopCount: number; interruptible: boolean }[]>([])
let actionTimer: number | null = null

// 空闲状态计时
const idleStartTime = ref(Date.now())
const lastActivityTime = ref(Date.now())
let idleCheckTimer: number | null = null

// 点击计数
const clickCount = ref(0)
const firstClickTime = ref(0)
const lastClickTime = ref(0)
const CLICK_WINDOW = 3000 // 3秒内的点击算连续点击

// running状态计时
const runningStartTime = ref(0)
const RUNNING_DURATION = 180000 // running持续3分钟
let runningCheckTimer: number | null = null

// 状态超时时间
const SITTING_TO_SLEEPING_MS = 60000 // 60秒无操作进入睡眠
const SITTING_TO_BORED_MS = 1800000 // 30分钟进入无聊
const TALKING_REMINDER_INTERVAL_MS = 3600000 // 1小时休息提醒

// 动画过渡时间
const TRANSITION_DURATION = 300

// ==================== 计算属性 ====================
export const currentGif = computed(() => {
  if (currentAction.value) {
    return PETS[currentPet.value][currentAction.value]
  }
  return PETS[currentPet.value][baseState.value]
})

// ==================== 动作队列管理 ====================
export function enqueueAction(action: TempAction, duration = 2000, loopCount = 1) {
  actionQueue.value.push({ name: action, duration, loop: loopCount })
  if (!currentAction.value) {
    processNextAction()
  }
}

// 清空低优先级动作（状态切换时调用）
function clearLowPriorityActions() {
  actionQueue.value = actionQueue.value.filter(a => 
    ['talking'].includes(a.name) // 保留高优先级动作
  )
}

function processNextAction() {
  if (actionTimer) {
    clearTimeout(actionTimer)
  }
  
  const next = actionQueue.value.shift()
  if (next) {
    currentAction.value = next.name
    
    // 循环播放支持（crazy动作循环3次）
    const actualDuration = next.duration * (next.loop || 1)
    
    actionTimer = window.setTimeout(() => {
      currentAction.value = null
      processNextAction()
    }, actualDuration)
  }
}

// ==================== 空闲状态递进逻辑 ====================
function checkIdleState() {
  // running状态下不进行空闲检测
  if (currentAction.value === 'running') return
  
  const now = Date.now()
  const idleTime = now - lastActivityTime.value
  
  // sitting → sleeping（60秒）
  if (baseState.value === 'sitting' && idleTime >= SITTING_TO_SLEEPING_MS) {
    baseState.value = 'sleeping'
    return
  }
  
  // sitting 超过30分钟触发 bored 状态
  if (baseState.value === 'sitting' && idleTime >= SITTING_TO_BORED_MS) {
    enqueueAction('bored', 2000, 'idle-transition')
    return
  }
  
  // 1小时连续使用触发休息提醒
  const timeSinceLastReminder = now - lastTalkingReminderTime.value
  if (timeSinceLastReminder >= TALKING_REMINDER_INTERVAL_MS) {
    enqueueLoopAction('talking', 2000, 3, 'instruction', false)
    lastTalkingReminderTime.value = now
    return
  }
}

// ==================== 点击互动逻辑 ====================
export function onPetClick() {
  const now = Date.now()
  
  // 更新活动时间，重置空闲计时
  lastActivityTime.value = now
  idleStartTime.value = now
  
  // 如果从sleeping状态恢复，先播放过渡动画
  if (baseState.value === 'sleeping' && currentAction.value !== 'running') {
    wakeFromIdle('lying-up', 1200)
    clickCount.value = 0
    return
  }
  
  // 3秒时间窗口内的连续点击计数
  if (now - firstClickTime.value > CLICK_WINDOW) {
    clickCount.value = 1
    firstClickTime.value = now
  } else {
    clickCount.value++
  }
  lastClickTime.value = now
  
  // 根据当前状态和点击次数触发不同动作
  handleClickInteraction()
}

function handleClickInteraction() {
  const clicks = clickCount.value
  
  if (currentAction.value === 'running') {
    // running状态下点击不触发新动作（让它冷静）
    return
  }
  
  if (baseState.value === 'sitting') {
    // sitting状态点击逻辑
    if (clicks === 2) {
      enqueueAction('tilt-head', 1500, 'interaction')
    } else if (clicks >= 5) {
      clearLowPriorityActions()
      enqueueAction('wag-tail', 1800, 'interaction')
    } else {
      enqueueAction('happy', 1500, 'interaction')
    }
  }
  
  if (baseState.value === 'sleeping') {
    if (clicks >= 50) {
      enqueueLoopAction('running', RUNNING_DURATION, 1, 'instruction', false)
      enqueueAction('backing', 1600, 'instruction')
      baseState.value = 'sitting'
      resetClickCounter()
      return
    }
    if (clicks >= 15) {
      enqueueLoopAction('crazy-plus', 1300, 2, 'interaction', true)
      return
    }
    if (clicks >= 5) {
      enqueueLoopAction('crazy', 1200, 3, 'interaction', true)
      return
    }
  }
}

// ==================== Running特殊状态 ====================
function enterRunningState() {
  currentAction.value = 'running'
  runningStartTime.value = Date.now()
  clickCount.value = 0
  
  // 启动恢复计时器
  if (runningCheckTimer) clearTimeout(runningCheckTimer)
  runningCheckTimer = window.setTimeout(() => {
    exitRunningState()
  }, RUNNING_DURATION)
}

function exitRunningState() {
  clearLowPriorityActions()
  enqueueAction('backing', 2000)
  
  // backing动作完成后，状态回到sitting
  setTimeout(() => {
    baseState.value = 'sitting'
    currentAction.value = null
    clickCount.value = 0
    processNextAction()
  }, 2000)
}
```

```rust
    idle_threshold: Duration,
    poll_interval: Duration,
    is_active: bool,

impl ActivityMonitor {
    pub fn new() -> Self {
        Self {
            idle_threshold: Duration::from_secs(30),
            poll_interval: Duration::from_secs(5),
            is_active: true,
        }
    }
    
    pub fn start(&self, window: Window) {
        let idle_threshold = self.idle_threshold;
        let poll_interval = self.poll_interval;
        let mut last_state = true;
        
        tauri::async_runtime::spawn(async move {
            loop {
                let idle_ms = get_idle_time_ms();
                let is_active = Duration::from_millis(idle_ms as u64) < idle_threshold;
                
                if is_active != last_state {
                    last_state = is_active;
                    // 向前端发送状态变更事件
                    window.emit("activity-state-changed", 
                        serde_json::json!({ "active": is_active })).unwrap();
                }
                
                tokio::time::sleep(poll_interval).await;
            }
        });
    }
}
```

pub fn create_engine(config: &AppConfig) -> Result<Box<dyn AsrEngine>, String> {
    match config.asr.provider.as_str() {
        "whisper-local" => {
            let model_path = format!("models/ggml-{}.bin", config.asr.whisper_local.model_size);
            let engine = WhisperLocalEngine::new(&model_path).map_err(|e| e.to_string())?;
            Ok(Box::new(engine))
        }
        // 所有在线ASR走统一的HTTP客户端逻辑
        "dashscope" | "volcengine" | "funasr" => {
            let cfg = match config.asr.provider.as_str() {
                "dashscope" => &config.asr.dashscope,
                "volcengine" => &config.asr.volcengine,
                "funasr" => &config.asr.funasr,
                _ => unreachable!(),
            };
            
            if cfg.base_url.is_empty() {
                return Err("请配置服务地址 Base URL".to_string());
            }
            
            // 统一的在线ASR引擎，api_key/model/base_url 直接透传
            let engine = OnlineAsrEngine::new(
                cfg.api_key.clone(),
                cfg.model.clone(),
                cfg.base_url.clone(),
            );
            Ok(Box::new(engine))
        }
        _ => Err(format!("不支持的ASR提供商: {}", config.asr.provider))

```rust
    Ollama { base_url: String, model: String },
}

pub struct LlmAgent {
    provider: LlmProvider,
}

impl LlmAgent {
    pub fn new(provider: LlmProvider) -> Self {
        Self { provider }
    }
    
    pub async fn chat(&self, user_input: &str) -> Result<(String, Option<String>), Box<dyn std::error::Error>> {
        let system_prompt = r#"
你是一只可爱的宠物狗助手。请根据用户的话给出回复。
回复JSON格式：{"text": "回复内容", "action": "动作名或null"}
动作可选：wag-tail / happy / lying-down / tilt-head
        "#;
        
        match &self.provider {
            LlmProvider::OpenAI { api_key, model } => {
                let client = Client::new().with_api_key(api_key);
                
                let request = vec![
                    ChatCompletionRequestMessageArgs::default()
                        .role(Role::System)
                        .content(system_prompt)
                        .build()?,
                    ChatCompletionRequestMessageArgs::default()
                        .role(Role::User)
                        .content(user_input)
                        .build()?,
                ];
                
                let response = client.chat().create(request).await?;
                let content = response.choices[0].message.content.as_ref().unwrap();
                
                // 解析JSON回复
                let parsed: Value = serde_json::from_str(content)?;
                let text = parsed["text"].as_str().unwrap_or(content).to_string();
                let action = parsed["action"].as_str().map(|s| s.to_string());
                
                Ok((text, action))
            }
        }
    }
}

pub async fn send_to_llm(user_input: String) -> Result<(String, Option<String>), String> {
    // 从配置读取provider
    let agent = LlmAgent::new(get_provider_from_config());
```

```rust
// src-tauri/src/skills/mod.rs

#[derive(Debug, Serialize, Deserialize)]
pub struct Skill {
    pub id: String,
    pub name: String,
    pub description: String,
    pub icon: String,
    pub enabled: bool,
    pub function: FunctionDefinition,
    pub runner: SkillRunner,
}

#[derive(Debug, Serialize, Deserialize)]
#[serde(tag = "type")]
pub enum SkillRunner {
    Http { url: String, method: String },  // 远程HTTP技能（skills.sh）
    Local { command: String },              // 本地命令行技能
    Builtin { handler: String },            // 内置Rust技能
}

pub struct SkillManager {
    skills: Vec<Skill>,
    installed_path: PathBuf,
}

impl SkillManager {
    /// 从 skills.sh 搜索/安装技能
    pub async fn install_from_shop(&mut self, skill_id: &str) -> Result<(), String> {
        // 1. 从 skills.sh 下载技能定义JSON
        // 2. 保存到本地 skills 目录
        // 3. 加入技能列表
    }
    
    /// 获取所有已启用技能的函数定义（传给LLM）
    pub fn get_function_definitions(&self) -> Vec<&FunctionDefinition> {
        self.skills
            .iter()
            .filter(|s| s.enabled)
            .map(|s| &s.function)
            .collect()
    }
    
    /// 执行技能
    pub async fn run_skill(&self, skill_id: &str, args: serde_json::Value) -> Result<String, String> {
        let skill = self.skills.iter().find(|s| s.id == skill_id).ok_or("技能不存在")?;
        
        match &skill.runner {
            SkillRunner::Http { url, method } => {
                // HTTP调用 skills.sh
                let client = reqwest::Client::new();
                let response = client
                    .request(method.parse().unwrap(), url)
                    .json(&args)
                    .send()
                    .await
                    .map_err(|e| e.to_string())?;
                
                response.text().await.map_err(|e| e.to_string())
            }
            SkillRunner::Local { command } => {
                // 执行本地命令
                let output = std::process::Command::new("sh")
                    .arg("-c")
                    .arg(command)
                    .output()
                    .map_err(|e| e.to_string())?;
                
                Ok(String::from_utf8_lossy(&output.stdout).to_string())
            }
            SkillRunner::Builtin { handler } => {
                // 内置技能，调用对应handler
                self.run_builtin_skill(handler, args).await
            }
        }
    }
}
```

```rust
impl LlmAgent {
    pub async fn chat_with_skills(
        &self, 
        user_input: &str,
        skill_manager: &SkillManager
    ) -> Result<ChatResponse, Box<dyn std::error::Error>> {
        
        // 1. 把所有已启用技能的定义加入prompt/tool_call
        let tools = skill_manager.get_function_definitions();
        
        // 2. 第一次调用LLM
        let response = self.call_llm_with_tools(user_input, tools).await?;
        
        // 3. 检查LLM是否要求调用工具
        if let Some(tool_call) = response.tool_calls.first() {
            // 4. 执行技能
            let skill_result = skill_manager
                .run_skill(&tool_call.function.name, &tool_call.function.arguments)
                .await?;
            
            // 5. 把技能结果传给LLM，生成最终回复
            let final_response = self.call_llm_with_result(user_input, &skill_result).await?;
            
            return Ok(final_response);
        }
        
        // 不需要调用技能，直接返回
        Ok(response)
    }
}
```
